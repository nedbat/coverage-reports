476 program_path = sys.argv[0]

524 """The bulk of the command line interface to coverage.py. 

525

527 

528 Returns 0 if all is well, 1 if something went wrong.

541 else:

699 def do_run(self, options, args): 

700 """Implementation of 'coverage run'.""" 

701 

702 if not args: 

703 if options.module: 

704 # Specified -m with nothing else. 

705 show_help("No module specified for -m") 

706 return ERR 

707 command_line = self.coverage.get_option("run:command_line") 

708 if command_line is not None: 

709 args = shlex.split(command_line) 

710 if args and args[0] == "-m": 

711 options.module = True 

712 args = args[1:] 

713 if not args: 

714 show_help("Nothing to do.") 

715 return ERR 

716 

717 if options.append and self.coverage.get_option("run:parallel"): 

718 show_help("Can't append to data files in parallel mode.") 

719 return ERR 

720 

721 if options.concurrency == "multiprocessing": 

722 # Can't set other run-affecting command line options with
