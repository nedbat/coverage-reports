108 def _check_arcs(self, a1, a2, arc_type): 

109 """Check that the arc lists `a1` and `a2` are equal. 

110 

111 If they are equal, return empty string. If they are unequal, return 

112 a string explaining what is different. 

113 """ 

114 # Make them into multi-line strings so we can see what's going wrong. 

115 s1 = arcs_to_arcz_repr(a1) 

116 s2 = arcs_to_arcz_repr(a2) 

117 if s1 != s2: 

118 lines1 = s1.splitlines(True) 

119 lines2 = s2.splitlines(True) 

120 diff = "".join(difflib.ndiff(lines1, lines2)) 

121 return "\n" + arc_type + " arcs differ: minus is expected, plus is actual\n" + diff 

122 else: 

123 return ""

125 def check_coverage( 

126 self, text, lines=None, missing="", report="", 

127 excludes=None, partials="", 

128 arcz=None, arcz_missing=None, arcz_unpredicted=None, 

129 arcs=None, arcs_missing=None, arcs_unpredicted=None, 

130 ): 

131 """Check the coverage measurement of `text`. 

132 

133 The source `text` is run and measured. `lines` are the line numbers 

134 that are executable, or a list of possible line numbers, any of which 

135 could match. `missing` are the lines not executed, `excludes` are 

136 regexes to match against for excluding lines, and `report` is the text 

137 of the measurement report. 

138 

139 For arc measurement, `arcz` is a string that can be decoded into arcs 

140 in the code (see `arcz_to_arcs` for the encoding scheme). 

141 `arcz_missing` are the arcs that are not executed, and 

142 `arcz_unpredicted` are the arcs executed in the code, but not deducible 

143 from the code. These last two default to "", meaning we explicitly 

144 check that there are no missing or unpredicted arcs. 

145 

146 Returns the Coverage object, in case you want to poke at it some more. 

147 

148 """ 

149 # We write the code into a file so that we can import it. 

150 # Coverage.py wants to deal with things as modules with file names. 

151 modname = self.get_module_name() 

152 

153 self.make_file(modname + ".py", text) 

154 

155 if arcs is None and arcz is not None: 

156 arcs = arcz_to_arcs(arcz) 

157 if arcs_missing is None and arcz_missing is not None: 

158 arcs_missing = arcz_to_arcs(arcz_missing) 

159 if arcs_unpredicted is None and arcz_unpredicted is not None: 

160 arcs_unpredicted = arcz_to_arcs(arcz_unpredicted) 

161 

162 # Start up coverage.py. 

163 cov = coverage.Coverage(branch=True) 

164 cov.erase() 

165 for exc in excludes or []: 

166 cov.exclude(exc) 

167 for par in partials or []: 

168 cov.exclude(par, which='partial') 

169 

170 mod = self.start_import_stop(cov, modname) 

171 

172 # Clean up our side effects 

173 del sys.modules[modname] 

174 

175 # Get the analysis results, and check that they are right. 

176 analysis = cov._analyze(mod) 

177 statements = sorted(analysis.statements) 

178 if lines is not None: 

179 if isinstance(lines[0], int): 

180 # lines is just a list of numbers, it must match the statements 

181 # found in the code. 

182 assert statements == lines, "{!r} != {!r}".format(statements, lines) 

183 else: 

184 # lines is a list of possible line number lists, one of them 

185 # must match. 

186 for line_list in lines: 

187 if statements == line_list: 

188 break 

189 else: 

190 self.fail("None of the lines choices matched %r" % statements) 

191 

192 missing_formatted = analysis.missing_formatted() 

193 if isinstance(missing, string_class): 

194 msg = "{!r} != {!r}".format(missing_formatted, missing) 

195 assert missing_formatted == missing, msg 

196 else: 

197 for missing_list in missing: 

198 if missing_formatted == missing_list: 

199 break 

200 else: 

201 self.fail("None of the missing choices matched %r" % missing_formatted) 

202 

203 if arcs is not None: 

204 # print("Possible arcs:") 

205 # print(" expected:", arcs) 

206 # print(" actual:", analysis.arc_possibilities()) 

207 # print("Executed:") 

208 # print(" actual:", sorted(set(analysis.arcs_executed()))) 

209 # TODO: this would be nicer with pytest-check, once we can run that. 

210 msg = ( 

211 self._check_arcs(arcs, analysis.arc_possibilities(), "Possible") + 

212 self._check_arcs(arcs_missing, analysis.arcs_missing(), "Missing") + 

213 self._check_arcs(arcs_unpredicted, analysis.arcs_unpredicted(), "Unpredicted") 

214 ) 

215 if msg: 

216 assert False, msg 

217 

218 if report: 

219 frep = StringIO() 

220 cov.report(mod, file=frep, show_missing=True) 

221 rep = " ".join(frep.getvalue().split("\n")[2].split()[1:]) 

222 assert report == rep, "{!r} != {!r}".format(report, rep) 

223 

224 return cov

226 @contextlib.contextmanager 

227 def assert_warnings(self, cov, warnings, not_warnings=()): 

228 """A context manager to check that particular warnings happened in `cov`. 

229 

230 `cov` is a Coverage instance. `warnings` is a list of regexes. Every 

231 regex must match a warning that was issued by `cov`. It is OK for 

232 extra warnings to be issued by `cov` that are not matched by any regex. 

233 Warnings that are disabled are still considered issued by this function. 

234 

235 `not_warnings` is a list of regexes that must not appear in the 

236 warnings. This is only checked if there are some positive warnings to 

237 test for in `warnings`. 

238 

239 If `warnings` is empty, then `cov` is not allowed to issue any 

240 warnings. 

241 

242 """ 

243 saved_warnings = [] 

244 def capture_warning(msg, slug=None, once=False): # pylint: disable=unused-argument 

245 """A fake implementation of Coverage._warn, to capture warnings.""" 

246 # NOTE: we don't implement `once`. 

247 if slug: 

248 msg = "%s (%s)" % (msg, slug) 

249 saved_warnings.append(msg) 

250 

251 original_warn = cov._warn 

252 cov._warn = capture_warning 

253 

254 try: 

255 yield 

256 except: # pylint: disable=try-except-raise 

257 raise 

258 else: 

259 if warnings: 

260 for warning_regex in warnings: 

261 for saved in saved_warnings: 

262 if re.search(warning_regex, saved): 

263 break 

264 else: 

265 self.fail("Didn't find warning %r in %r" % (warning_regex, saved_warnings)) 

266 for warning_regex in not_warnings: 

267 for saved in saved_warnings: 

268 if re.search(warning_regex, saved): 

269 self.fail("Found warning %r in %r" % (warning_regex, saved_warnings)) 

270 else: 

271 # No warnings expected. Raise if any warnings happened. 

272 if saved_warnings: 

273 self.fail("Unexpected warnings: %r" % (saved_warnings,)) 

274 finally: 

275 cov._warn = original_warn

354 def run_command_status(self, cmd): 

355 """Run the command-line `cmd` in a sub-process, and print its output. 

356 

357 Use this when you need to test the process behavior of coverage. 

358 

359 Compare with `command_line`. 

360 

361 Handles the following command names specially: 

362 

363 * "python" is replaced with the command name of the current 

364 Python interpreter. 

365 

366 * "coverage" is replaced with the command name for the main 

367 coverage.py program. 

368 

369 Returns a pair: the process' exit status and its stdout/stderr text, 

370 which are also stored as `self.last_command_status` and 

371 `self.last_command_output`. 

372 

373 """ 

374 # Make sure "python" and "coverage" mean specifically what we want 

375 # them to mean. 

376 split_commandline = cmd.split() 

377 command_name = split_commandline[0] 

378 command_args = split_commandline[1:] 

379 

380 if command_name == "python": 

381 # Running a Python interpreter in a sub-processes can be tricky. 

382 # Use the real name of our own executable. So "python foo.py" might 

383 # get executed as "python3.3 foo.py". This is important because 

384 # Python 3.x doesn't install as "python", so you might get a Python 

385 # 2 executable instead if you don't use the executable's basename. 

386 command_words = [os.path.basename(sys.executable)] 

387 

388 elif command_name == "coverage": 

389 if env.JYTHON: # pragma: only jython 

390 # Jython can't do reporting, so let's skip the test now. 

391 if command_args and command_args[0] in ('report', 'html', 'xml', 'annotate'): 

392 pytest.skip("Can't run reporting commands in Jython") 

393 # Jython can't run "coverage" as a command because the shebang 

394 # refers to another shebang'd Python script. So run them as 

395 # modules. 

396 command_words = "jython -m coverage".split() 

397 else: 

398 # The invocation requests the coverage.py program. Substitute the 

399 # actual coverage.py main command name. 

400 command_words = [self.coverage_command] 

401 

402 else: 

403 command_words = [command_name] 

404 

405 cmd = " ".join([shlex_quote(w) for w in command_words] + command_args) 

406 

407 # Add our test modules directory to PYTHONPATH. I'm sure there's too 

408 # much path munging here, but... 

409 pythonpath_name = "PYTHONPATH" 

410 if env.JYTHON: 

411 pythonpath_name = "JYTHONPATH" # pragma: only jython 

412 

413 testmods = self.nice_file(self.working_root(), 'tests/modules') 

414 zipfile = self.nice_file(self.working_root(), 'tests/zipmods.zip') 

415 pypath = os.getenv(pythonpath_name, '') 

416 if pypath: 

417 pypath += os.pathsep 

418 pypath += testmods + os.pathsep + zipfile 

419 self.set_environ(pythonpath_name, pypath) 

420 

421 self.last_command_status, self.last_command_output = run_command(cmd) 

422 print(self.last_command_output) 

423 return self.last_command_status, self.last_command_output
